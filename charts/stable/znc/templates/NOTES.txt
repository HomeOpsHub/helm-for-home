    kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "znc.fullname" . }} 6501:{{ .Values.service.port }}

  View logs:
    kubectl logs --namespace {{ .Release.Namespace }} -l app.kubernetes.io/instance={{ .Release.Name }}
