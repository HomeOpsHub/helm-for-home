Thank you for installing {{ .Chart.Name }} v{{ .Chart.AppVersion }}!

    kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "loki.fullname" . }} 3100:{{ .Values.service.port }}

  Check pod status:
    kubectl get pods --namespace {{ .Release.Namespace }} -l app.kubernetes.io/instance={{ .Release.Name }}

  View logs:
    kubectl logs --namespace {{ .Release.Namespace }} -l app.kubernetes.io/instance={{ .Release.Name }}
